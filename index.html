#!/bin/sh ############################################################################### ############################################################################### ############################################################################### ################################################# https://github.com/mxcl/yoink ############################################################################### ############################################################################### ############################################################################### tmp=$(mktemp -d) trap 'rm -rf "$tmp"' EXIT curl -Ssf "https://github.com/mxcl/yoink/releases/v0.1.0/download/yoink-0.1.0-$(uname -s)-$(uname -m).tar.gz" | tar -xz -C "$tmp" "$tmp/yoink" "$@"
